function processAndRender(rows) { const today = new Date(); today.setHours(0, 0, 0, 0); // Reset time for accurate date comparison // Initializing structure for Branch mapping const summary = { "East Branch": { morning: 0, afternoon: 0, evening: 0 }, "West Branch": { morning: 0, afternoon: 0, evening: 0 } }; rows.forEach(row => { // Step 1: Check Expiry Date (Column F: Access Expiry (DD-MM-YY)) const expiryStr = row["Access Expiry (DD-MM-YY)"]; if (!expiryStr || expiryStr === "--") return; const [day, month, year] = expiryStr.split('-').map(Number); const expiryDate = new Date(2000 + year, month - 1, day); // Only process if the subscription is NOT expired if (expiryDate >= today) { // Step 2: Determine Branch based on Center (Column L) let branchKey = ""; const centerId = row["Center"]; if (centerId === "SZ_04") { branchKey = "East Branch"; } else if (centerId === "SZ_01" || centerId === "SZ_03") { branchKey = "West Branch"; } // Step 3: Categorize Slot based on Plan (Column N) if (branchKey) { const plan = row["Plan"] || ""; // Morning Slots Logic if (plan.includes("Morning") || plan.includes("Full Day") || plan.includes("Day Plan")) { summary[branchKey].morning++; } // Afternoon Slots Logic if (plan.includes("Afternoon") || plan.includes("Full Day") || plan.includes("Day Plan")) { summary[branchKey].afternoon++; } // Evening Slots Logic if (plan.includes("Evening") || plan.includes("Full Day") || plan.includes("Day Plan")) { summary[branchKey].evening++; } } } }); renderUI(summary); } function renderUI(summary) { const container = document.getElementById('branchContainer'); container.innerHTML = ""; for (const [branchName, counts] of Object.entries(summary)) { const isEast = branchName.includes("East"); const slots = [ { label: "Morning", val: counts.morning }, { label: "Afternoon", val: counts.afternoon }, { label: "Evening", val: counts.evening } ]; const branchHtml = `

${slots.map(s => { // Set 60 as the base capacity for the progress bar calculation const percent = Math.min((s.val / 60) * 100, 100); return `
${s.label} ${s.val} Students
`; }).join('')}
`; container.insertAdjacentHTML('beforeend', branchHtml); } }
