processAndRender(results.data); } else { document.getElementById('loading').innerText = "Error: No data found in CSV."; } }, error: function(err) { document.getElementById('loading').innerText = "Error loading CSV: " + err; } }); function processAndRender(rows) { const today = new Date(); today.setHours(0, 0, 0, 0); // Initializing structure for Branch mapping const summary = { "East Branch": { morning: 0, afternoon: 0, evening: 0 }, "West Branch": { morning: 0, afternoon: 0, evening: 0 } }; rows.forEach(row => { // Step 1: Check Expiry Date (Column F) const expiryStr = row["Access Expiry (DD-MM-YY)"]; if (!expiryStr || expiryStr.trim() === "" || expiryStr === "--") return; const parts = expiryStr.split('-'); if (parts.length !== 3) return; const [day, month, year] = parts.map(Number); const expiryDate = new Date(2000 + year, month - 1, day); // Only process if the subscription is NOT expired if (expiryDate >= today) { // Step 2: Determine Branch based on Center (Column L) let branchKey = ""; const centerId = (row["Center"] || "").trim(); if (centerId === "SZ_04") { branchKey = "East Branch"; } else if (centerId === "SZ_01" || centerId === "SZ_03") { branchKey = "West Branch"; } // Step 3: Categorize Slot based on Plan (Column N) if (branchKey) { const plan = (row["Plan"] || "").toLowerCase(); // Morning Logic: Includes Morning Batch, Morning+Afternoon, Morning+Evening, Full Day if (plan.includes("morning") || plan.includes("full day") || plan.includes("day plan")) { summary[branchKey].morning++; } // Afternoon Logic: Includes Afternoon Batch, Afternoon+Evening, Morning+Afternoon, Full Day if (plan.includes("afternoon") || plan.includes("full day") || plan.includes("day plan")) { summary[branchKey].afternoon++; } // Evening Logic: Includes Evening Batch, Morning+Evening, Afternoon+Evening, Full Day if (plan.includes("evening") || plan.includes("full day") || plan.includes("day plan")) { summary[branchKey].evening++; } } } }); renderUI(summary); } function renderUI(summary) { const container = document.getElementById('branchContainer'); if (!container) return; container.innerHTML = ""; for (const [branchName, counts] of Object.entries(summary)) { const isEast = branchName.includes("East"); const slots = [ { label: "Morning", val: counts.morning }, { label: "Afternoon", val: counts.afternoon }, { label: "Evening", val: counts.evening } ]; const branchHtml = `

`; }).join('')}
`; container.insertAdjacentHTML('beforeend', branchHtml); } }
