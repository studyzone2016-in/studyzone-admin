const expiryDate = new Date(2000 + year, month - 1, day); // Only proceed if subscription is active (Expiry >= Today) if (expiryDate >= today) { // Logic B: Determine Branch (Column L - Center) let branch = ""; const centerId = row["Center"]; if (centerId === "SZ_04") { branch = "East Branch"; } else if (centerId === "SZ_01" || centerId === "SZ_03") { branch = "West Branch"; } // Logic C: Analyze Plan (Column N - Plan) const plan = row["Plan"] || ""; if (branch) { // Morning Check if (plan.includes("Morning") || plan.includes("Full Day")) { summary[branch].morning++; } // Afternoon Check if (plan.includes("Afternoon") || plan.includes("Full Day")) { summary[branch].afternoon++; } // Evening Check if (plan.includes("Evening") || plan.includes("Full Day")) { summary[branch].evening++; } } } }); renderUI(summary); } function renderUI(summary) { const container = document.getElementById('branchContainer');
